% --- Paragraph layout: no indent, small fixed gap (no stretch, so spacing is
%     identical everywhere) ----------------------------------------------------
\setlength{\parindent}{0pt}
\setlength{\parskip}{4pt}

% --- Colors ------------------------------------------------------------------
\definecolor{accent}{RGB}{14, 84, 78}

% --- Hyperlink styling ---------------------------------------------------------
\hypersetup{
    colorlinks=true,
    urlcolor=accent,
    linkcolor=accent,
}

% --- Section headers: colored caps with a rule underneath --------------------
\titleformat{\section}
  {\large\bfseries\color{accent}}
  {}{0em}{}
  [\vspace{2pt}{\color{accent}\titlerule[0.8pt]}]
\titlespacing*{\section}{0pt}{7pt}{3pt}

% --- Tighter lists for bullets ------------------------------------------------
\setlist[itemize]{leftmargin=1.15em, itemsep=1pt, topsep=2pt, parsep=0pt}

% --- No page numbers -----------------------------------------------------------
\pagestyle{empty}

% --- Avoid overfull lines with long bullet text, while keeping justified text
%     tidy (small emergency stretch instead of full \sloppy, which can leave
%     large, unprofessional gaps between words) --------------------------------
\setlength{\emergencystretch}{2em}

% --- Helper command for an experience/education header row -------------------
% #1 = left bold text, #2 = right-aligned text (e.g. dates)
\newcommand{\resumeRow}[2]{%
  \noindent\begin{tabularx}{\linewidth}{@{}X r@{}}
    \textbf{#1} & #2 \\
  \end{tabularx}%
}

% --- Two-column bulleted list (for Certifications, Awards, Interests) --------
% #1 = list of \item entries for the left column, #2 = same for the right column
\newenvironment{twocolitems}{%
  \begin{itemize}[leftmargin=1.15em, itemsep=1pt, topsep=2pt, parsep=0pt]
}{%
  \end{itemize}
}
\newcommand{\twocollist}[2]{%
  \noindent\begin{minipage}[t]{0.5\linewidth}
    \begin{twocolitems}
    #1
    \end{twocolitems}
  \end{minipage}%
  \begin{minipage}[t]{0.5\linewidth}
    \begin{twocolitems}
    #2
    \end{twocolitems}
  \end{minipage}%
}
